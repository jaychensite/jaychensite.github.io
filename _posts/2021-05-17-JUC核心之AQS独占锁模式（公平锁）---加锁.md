```java
        private static final long serialVersionUID = 7316153563782823691L;

        /**
         * Performs lock.  Try immediate barge, backing up to normal
         * acquire on failure.
         */
        final void lock() {
            if (compareAndSetState(0, 1))
                setExclusiveOwnerThread(Thread.currentThread());
            else
                acquire(1);

            return nonfairTryAcquire(acquires);
        }

static final class FairSync extends Sync {
        private static final long serialVersionUID = -3000897897090466540L;

        final void lock() {
            acquire(1);
        }

        /**
         * Fair version of tryAcquire.  Don't grant access unless
         * recursive call or no waiters or is first.
         */
        protected final boolean tryAcquire(int acquires) {
            final Thread current = Thread.currentThread();
            int c = getState();
            if (c == 0) {
                if (!hasQueuedPredecessors() &&
                    compareAndSetState(0, acquires)) {
                    setExclusiveOwnerThread(current);
                    return true;
                }
            }
            else if (current == getExclusiveOwnerThread()) {
                int nextc = c + acquires;
                if (nextc < 0)
                    throw new Error("Maximum lock count exceeded");
                setState(nextc);
                return true;
            }
            return false;
        }
 }
```

```java
// 加锁入口
public void lock() {
  // 调用sync的lock对象。通过你创建的ReentrantLock来决定调用公平锁的lock具体实现还是非公平锁的具体实现
  sync.lock();

// FairSync.lock
final void lock() {
  // 获取锁，注意这里是AQS的acquire
  acquire(1);
}

public final void acquire(int arg) {
  			// 条件1:tryAcquire尝试获取锁，如果返回false则条件不成立。
  			// 条件2:acquireQueued（addWaiter(Node.EXCLUSIVE), arg)）返回true:表示线程再挂起过程中，中断唤醒过线程，否则没有。
  			//     2.1: addWaiter(Node.EXCLUSIVE)入队列
  			//     2.2:acquireQueued挂起线程
        if (!tryAcquire(arg) &&
            acquireQueued(addWaiter(Node.EXCLUSIVE), arg))
          	// 中断线程
            selfInterrupt();
}

// tryAcquire是AQS的方法，FairSync实现了tryAcquire
protected final boolean tryAcquire(int acquires) {
  					// 获取当前线程
            final Thread current = Thread.currentThread();
  					// 获取当前锁状态
            int c = getState();
  					// 如果c == 0 则说明此时处于无锁状态
            if (c == 0) {
              	// 条件1:hasQueuedPredecessors判断是否有人在等待锁
              	// 条件2:compareAndSetState CAS加锁
                if (!hasQueuedPredecessors() &&
                    compareAndSetState(0, acquires)) {
                  	// 加锁成功后则设置当前持有锁的线程为当前线程
                  	// 注意此时只是设置了当前持有锁的线程，并没有初始化node队列
                    setExclusiveOwnerThread(current);
                  	// 返回ture表示成功获取到锁
                    return true;
                }
            }
  					// 如果此时锁处于加锁状态，判断持有锁的线程是否是自己（这里就是重入锁的实现）
            else if (current == getExclusiveOwnerThread()) {
             		// 累计
                int nextc = c + acquires;
              	// 这里会判断累计的大小超过int值最大范围
                if (nextc < 0)
                    throw new Error("Maximum lock count exceeded");
              	// set值，这里不需要CAS，因为当前线程已经持有锁了。
                setState(nextc);
              	// true加锁成功
                return true;
            }
  					// 加锁失败
  					// 有哪几种情况会来到这里？？？
  					// 1:无锁状态下加锁失败
  					// 2:加锁状态下，持有锁的线程不是当前线程
            return false;
}
```

如果tryAcquire获取锁失败，则会进入acquireQueued(addWaiter(Node.EXCLUSIVE), arg)

```java
// 首先会将线程入队列 
private Node addWaiter(Node mode) {
  			// 为当前线程创建Node节点
        Node node = new Node(Thread.currentThread(), mode);
        // Try the fast path of enq; backup to full enq on failure
  			// 尾节点
        Node pred = tail;
  			// 如果尾节点不为null，则说明node队列已经初始化过
        if (pred != null) {
          	// 将尾节点复制给当前线程节点的前置节点
            node.prev = pred;
          	// CAS设置，如果失败则说明tail节点被其他线程所改变
            if (compareAndSetTail(pred, node)) {
              	// 设置当前尾节点的后继节点为当前节点
                pred.next = node;
              	// 返回当前节点
                return node;
            }
        }
  			// 如果入队列失败，或者 tail != null
        enq(node);
        return node;
}

private Node enq(final Node node) {
  			// 轮询
        for (;;) {
          	// tail临时变量
            Node t = tail;
          	// 如果尾节点为null
            if (t == null) { // Must initialize
              	// 必须初始化一个头节点
              	// 表示当前线程很有可能是第一个抢占锁失败的线程。前面源码分析过，线程抢占锁成功后，并没有设置头节点。
              	// 所以这里会尝试设置头节点
                if (compareAndSetHead(new Node()))
                  	// 将head节点赋值与尾节点
                  	// 注意这里并没有return;会继续轮询
                    tail = head;
            } else {
              	// 继续轮询时，会来到这里。
              	// 赋值给当前node的前继节点
                node.prev = t;
              	// CAS设置尾节点
                if (compareAndSetTail(t, node)) {
                    t.next = node;
                  	// return结束轮询
                    return t;
                }
            }
        }
}


// addWaiter里面只是包装了node，并入队列，但是我们并没有将线程挂起，也没有设置线程唤醒后的操作
// 以上操作则在acquireQueued中干的事情
// node：当前线程被包装成的Node节点
// arg：当前线程如果抢占锁成功后，会设置state。
final boolean acquireQueued(final Node node, int arg) {
        boolean failed = true;
        try {
          	// 是否中断标记
            boolean interrupted = false;
            for (;;) {
              	// 获取当前节点的前继节点
                final Node p = node.predecessor();
                // p == head ：如果p节点为头节点，当前节点就会去尝试获取锁（ps:任何情况下，head.next节点都有权限去尝试获取锁）
              	// tryAcquire
              	// 成立:说明对应的head节点已经释放锁了，刚好获取到锁了。
              	// 失败：说明对应的head节点还没释放锁，还是需要park当前线程
                if (p == head && tryAcquire(arg)) {
                  	// 如果获取到锁后，设置头节点为自己
                    setHead(node);
                  	// 将old head与后继节点断开引用。帮助old head出队列。
                    p.next = null; // help GC
                    failed = false;
                    return interrupted;
                }
                // shouldParkAfterFailedAcquire 当前线程获取锁失败后，是否需要挂起。true：挂起
              	// parkAndCheckInterrupt：把当前线程挂起
                if (shouldParkAfterFailedAcquire(p, node) &&
                    parkAndCheckInterrupt())
                    // 中断标记
                    interrupted = true;
            }
        } finally {
            // 一般情况下不会走到这里。
            // 如果等待过程中没有成功获取资源（如timeout，或者可中断的情况下被中断了），那么取消结点在队列中的等待。
            if (failed)
              	// 取消锁等待
                cancelAcquire(node);
        }
}

private static boolean shouldParkAfterFailedAcquire(Node pred, Node node) {
  			// 获取前继节点的状态
  			// waitStatus
  			// 0:默认状态，new code();
        // -1:signal状态，表示当前节点释放锁之后会唤醒后继节点。
  			// >0:表示当前节点为取消状态
        int ws = pred.waitStatus;
  		  // 如果前置节点为唤醒状态，则直接return
        if (ws == Node.SIGNAL)
            /*
             * This node has already set status asking a release
             * to signal it, so it can safely park.
             */
          	// 注意：前面的轮询，始终会进入到这里
            return true;
  			// 条件成立的情况下：表示前置节点取消了锁竞争
        if (ws > 0) {
            /*
             * Predecessor was cancelled. Skip over predecessors and
             * indicate retry.
             */
          	// 循环查找节点等待状态  <= 0;作为前置节点
            do {
                node.prev = pred = pred.prev;
            } while (pred.waitStatus > 0);
            pred.next = node;
        } else {
          	// 如果等于0，则强制将前置节点的等待状态改为唤醒后继节点状态
            /*
             * waitStatus must be 0 or PROPAGATE.  Indicate that we
             * need a signal, but don't park yet.  Caller will need to
             * retry to make sure it cannot acquire before parking.
             */
            compareAndSetWaitStatus(pred, ws, Node.SIGNAL);
        }
        return false;
}

// 取消锁等待，当线程中断时，会进行锁取消等待操作
private void cancelAcquire(Node node) {
        // Ignore if node doesn't exist
        if (node == null)
            return;
				// 将当前node节点thread重置为null。
        node.thread = null;

        // Skip cancelled predecessors
  			// 获取前继节点
        Node pred = node.prev;
       	// 条件成立：则说明前置节点也为取消状态，则循环找到不为取消的节点作为当前节点的前继节点
        while (pred.waitStatus > 0)
            node.prev = pred = pred.prev;

        // predNext is the apparent node to unsplice. CASes below will
        // fail if not, in which case, we lost race vs another cancel
        // or signal, so no further action is necessary.
  			// 前继节点的后继节点
        Node predNext = pred.next;

        // Can use unconditional write instead of CAS here.
        // After this atomic step, other Nodes can skip past us.
        // Before, we are free of interference from other threads.
  			// 将当前节点的状态修改为取消状态
        node.waitStatus = Node.CANCELLED;

        // If we are the tail, remove ourselves.
  			// 如果当前节点是尾节点，则直接移除，并将前置节点的后继节点设置为null
        if (node == tail && compareAndSetTail(node, pred)) {
            compareAndSetNext(pred, predNext, null);
        } else {
            // If successor needs signal, try to set pred's next-link
            // so it will get one. Otherwise wake it up to propagate.
          	// 前置节点的状态
            int ws;
          	// 条件1：pred != head -> 前置节点不为head节点
          	// 条件2.1：(ws = pred.waitStatus) == Node.SIGNAL -> 前置节点的状态是否为唤醒状态
          	// 条件2.2：(ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL)) -> 如果前置节点状态 <= 0,则CAS将												其修改为唤醒状态，为了唤醒后续节点而需要
          	// 条件3：pred.thread != null -> 前置节点的线程不能为null
            if (pred != head &&
                ((ws = pred.waitStatus) == Node.SIGNAL ||
                 (ws <= 0 && compareAndSetWaitStatus(pred, ws, Node.SIGNAL))) &&
                pred.thread != null) {
              	// 当前节点的next 节点
                Node next = node.next;
              	// 
                if (next != null && next.waitStatus <= 0)
                  	// 将前继节点的next 节点修改为当前节点的next节点
                    compareAndSetNext(pred, predNext, next);
            } else {
              	// 唤醒后继节点
                unparkSuccessor(node);
            }
						// 断开与next的引用
            node.next = node; // help GC
        }
}

// 唤醒后继节点
private void unparkSuccessor(Node node) {
        /*
         * If status is negative (i.e., possibly needing signal) try
         * to clear in anticipation of signalling.  It is OK if this
         * fails or if status is changed by waiting thread.
         */
        // 当前节点状态
        int ws = node.waitStatus;
  			// 条件成立则需要更改当前节点的状态
  			// <0 :表示当前线程 即可能需要前置节点的唤醒信号。但是我们这里已经将当前节点取消锁的等待，则不需要前置节点的唤醒等操作，修								改为默认状态
        if (ws < 0)
            compareAndSetWaitStatus(node, ws, 0);

        /*
         * Thread to unpark is held in successor, which is normally
         * just the next node.  But if cancelled or apparently null,
         * traverse backwards from tail to find the actual
         * non-cancelled successor.
         */
  			// 找到合适的后继节点，将起唤醒（也就是找到等待唤醒的后继节点）
        Node s = node.next;
        if (s == null || s.waitStatus > 0) {
            s = null;
            for (Node t = tail; t != null && t != node; t = t.prev)
                if (t.waitStatus <= 0)
                    s = t;
        }
        if (s != null)
          	// 唤醒
            LockSupport.unpark(s.thread);
}
```
